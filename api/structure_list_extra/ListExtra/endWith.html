bool endWith(List<T> other, {bool Function(T, T)? compare}) {
  if (other.length > this.length) {
    return false;
  }

  for (int i = 0; i < other.length; ++i) {
    var n = other[other.length - i - 1];
    var m = this[length - i - 1];
    if (!compare!(n, m)) {
      return false;
    }
  }

  return true;
}
