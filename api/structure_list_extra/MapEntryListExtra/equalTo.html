bool equalTo(List<MapEntry<K, V>> other) {
  if (this.length != other.length) {
    return false;
  }

  for (int i = 0; i < this.length; ++i) {
    if (!this[i].equalTo(other[i])) {
      return false;
    }
  }

  return true;
}
