(String?, String?) apartReverseBy(String gap) {
  final int pos = this.lastIndexOf(gap);
  if (pos == -1) {
    return (null, this);
  }
  final String first = this.substring(0, pos);
  final String last = this.substring(pos + gap.length);
  return (first, last);
}
